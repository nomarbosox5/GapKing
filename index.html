No games scheduled today.

`; card.onclick = () => startWatching(g); grid.appendChild(card); }); } catch (e) { console.error("Schedule Load Error:", e); } } async function startWatching(game) { selectedGamePk = game.gamePk; selectedTeamName = game.teams.home.team.name; document.getElementById('w-team-name').textContent = selectedTeamName; document.getElementById('screen-pick').classList.remove('active'); document.getElementById('screen-watch').classList.add('active'); // Reset UI document.getElementById('n-mlb').textContent = '...'; await updateOdds(); // Get odds first updateGumbo(); // Then start tracking gumboTimer = setInterval(updateGumbo, 2000); // 2s is safer for rate limits oddsTimer = setInterval(updateOdds, 60000); } async function updateGumbo() { try { const r = await fetch(`https://statsapi.mlb.com/api/v1.1/game/${selectedGamePk}/feed/live`); const d = await r.json(); // MLB Gumbo path for win probability const homeWinProb = d.liveData?.plays?.currentPlay?.about?.homeWinProbability ?? 50; const bookProb = calculateImpliedProb(lastOddsPrice); const gap = homeWinProb - bookProb; document.getElementById('n-mlb').textContent = homeWinProb.toFixed(1) + '%'; document.getElementById('n-book').textContent = bookProb.toFixed(1) + '%'; document.getElementById('n-gap').textContent = (gap > 0 ? '+' : '') + gap.toFixed(1) + '%'; document.getElementById('b-mlb').style.width = homeWinProb + '%'; document.getElementById('p-mlb').textContent = homeWinProb.toFixed(1) + '%'; document.getElementById('b-book').style.width = bookProb + '%'; document.getElementById('p-book').textContent = bookProb.toFixed(1) + '%'; const sigBox = document.getElementById('sig-box'); if (gap >= THRESH) { sigBox.className = 'sig-box buy'; document.getElementById('sig-lbl').textContent = '🚨 GAP DETECTED'; document.getElementById('sig-action').textContent = `BET ${selectedTeamName}`; } else { sigBox.className = 'sig-box idle'; document.getElementById('sig-lbl').textContent = 'SCANNING FEED'; document.getElementById('sig-action').textContent = '1s Gumbo Scan Active'; } } catch (e) { console.error("Gumbo Update Error:", e); } } async function updateOdds() { try { // We fetch the latest odds for MLB. Note: The-Odds-API supports CORS. const r = await fetch(`https://api.the-odds-api.com/v4/sports/baseball_mlb/odds/?apiKey=${ODDS_KEY}®ions=us&markets=h2h`); const d = await r.json(); // Find the specific game const game = d.find(o => o.home_team === selectedTeamName || o.away_team === selectedTeamName); if (game && game.bookmakers.length > 0) { const outcome = game.bookmakers[0].markets[0].outcomes.find(out => out.name === selectedTeamName); if (outcome) { lastOddsPrice = outcome.price; // This is usually decimal or American depending on API settings } } } catch (e) { console.error("Odds Update Error:", e); } } // Logic for American Odds to Prob function calculateImpliedProb(price) { if (price > 0) { return (100 / (price + 100)) * 100; } else { return (Math.abs(price) / (Math.abs(price) + 100)) * 100; } } function goBack() { clearInterval(gumboTimer); clearInterval(oddsTimer); document.getElementById('screen-watch').classList.remove('active'); document.getElementById('screen-pick').classList.add('active'); fetchSchedule(); } fetchSchedule();
